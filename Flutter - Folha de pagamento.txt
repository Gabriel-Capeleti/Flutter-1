      home: new MyHomePage(),

class MyHomePage extends StatefulWidget {
  MyHomePage({Key key}) : super(key: key);
  @override
  _MyHomePageState createState() => new _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  TextEditingController vHora= TextEditingController();
  TextEditingController qHora= TextEditingController();  
  TextEditingController sBruto= TextEditingController();  
    
    @override
    Widget build(BuildContext context) {
      return new Scaffold(
        appBar: new AppBar(
          title: new Text('Folha de Pagamento'),
          ),
        body:
          new Column(
            mainAxisAlignment: MainAxisAlignment.start,
            mainAxisSize: MainAxisSize.max,
            crossAxisAlignment: CrossAxisAlignment.center,
            children: <Widget>[
              new Icon(
                Icons.account_circle,
                color: const Color(0xFF241aee),
                size: 65.0),
    
              new Text(
              "Valor da Hora:",
                style: new TextStyle(fontSize:18.0,
                color: const Color(0xFF2613dd),
                fontWeight: FontWeight.w400,
                fontFamily: "Roboto"),
              ),
    
              new Padding(
                child:
                  new TextField(
                    style: new TextStyle(fontSize:18.0,
                    color: const Color(0xFFea651e),
                    fontWeight: FontWeight.w400,
                    fontFamily: "Roboto"),
                    controller: vHora,
                  ),
    
                padding: const EdgeInsets.fromLTRB(24.0, 0.0, 24.0, 24.0),
              ),
    
              new Text(
              "Quantidade de Hora:",
                style: new TextStyle(fontSize:18.0,
                color: const Color(0xFF1434e6),
                fontWeight: FontWeight.w400,
                fontFamily: "Roboto"),
              ),
    
              new Padding(
                child:
                  new TextField(
                    style: new TextStyle(fontSize:18.0,
                    color: const Color(0xFF1bdf2d),
                    fontWeight: FontWeight.w400,
                    fontFamily: "Roboto"),
                    controller: qHora,
                  ),
    
                padding: const EdgeInsets.fromLTRB(24.0, 0.0, 24.0, 24.0),
              ),
    
              new Text(
              "Salário Bruto:",
                style: new TextStyle(fontSize:18.0,
                color: const Color(0xFF2610f0),
                fontWeight: FontWeight.w400,
                fontFamily: "Roboto"),
              ),
    
              new Padding(
                child:
                  new TextField(
                    style: new TextStyle(fontSize:18.0,
                    color: const Color(0xFFf40b3a),
                    fontWeight: FontWeight.w400,
                    fontFamily: "Roboto"),
                    controller: sBruto,
                  ),
    
                padding: const EdgeInsets.fromLTRB(24.0, 0.0, 24.0, 24.0),
              ),
    
              new RaisedButton(key:null, onPressed:buttonPressed,
                color: const Color(0xFFe0e0e0),
                child:
                  new Text(
                  "Calcular",
                    style: new TextStyle(fontSize:18.0,
                    color: const Color(0xFF0f2edd),
                    fontWeight: FontWeight.w400,
                    fontFamily: "Roboto"),
                  )
                )
            ]
    
          ),
    
      );
    }
    void buttonPressed()
    {
      double vh , qh , sb;
		vh = double.parse(vHora.text);
		qh = double.parse(qHora.text);
		sb = vh * qh;
		sBruto.text = sb.toString();
    }
    
}
